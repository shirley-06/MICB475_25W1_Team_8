# Modify manifest to be appropriate for single end analysis
awk -v OFS='\t' 'NR==1{print "sample-id","absolute-filepath"; next} {print $1,$2}' \ /datasets/project_2/fermentation/fermentation_manifest.tsv > manifest_se.tsv

# Import reads into demux file
qiime tools import \
  --type 'SampleData[SequencesWithQuality]' \
  --input-path manifest_se.tsv \
  --input-format SingleEndFastqManifestPhred33V2 \
  --output-path demux_se.qza
  
# trim the forward primer (515F) at the 5' end of R1
qiime cutadapt trim-single \
  --i-demultiplexed-sequences demux_se.qza \
  --p-front GTGCCAGCMGCCGCGGTAA \
  --p-match-read-wildcards \
  --p-match-adapter-wildcards \
  --p-discard-untrimmed \
  --o-trimmed-sequences demux_se_trim.qza \
  --verbose > cutadapt_single.log  

# Convert into .qzv
qiime demux summarize \
--i-data demux_se_trim.qza \
--o-visualization demux_se_trim.qzv

# DADA2 single-end
qiime dada2 denoise-single \
  --i-demultiplexed-seqs demux_se_trim.qza \
  --p-trim-left 0 \
  --p-trunc-len 0 \
  --p-max-ee 2 \
  --p-n-threads 10 \
  --o-table table_se.qza \
  --o-representative-sequences rep_seqs_se.qza \
  --o-denoising-stats dada2_stats_se.qza

# Visualize ASVs stats
qiime feature-table summarize \
  --i-table table_se.qza \
  --o-visualization table_se_meta.qzv \
  --m-sample-metadata-file /datasets/project_2/fermentation/fermentation_metadata.tsv

 qiime metadata tabulate \
--m-input-file dada2_stats_se.qza \
--o-visualization dada2_stats_se.qzv

# Taxonomy
qiime feature-classifier classify-sklearn \
  --i-classifier /datasets/classifiers/silva-138-99-515-806-nb-classifier.qza \
  --i-reads rep_seqs_se.qza \
  --o-classification taxonomy_se.qza

qiime metadata tabulate \
  --m-input-file taxonomy_se.qza \
  --o-visualization taxonomy_se.qzv

qiime taxa barplot \
  --i-table table_se.qza \
  --i-taxonomy taxonomy_se.qza \
  --m-metadata-file /datasets/project_2/fermentation/fermentation_metadata.tsv \
  --o-visualization taxa_bar_plot_se.qzv

#Filtering out mitochondria and chloroplast
qiime taxa filter-table \
  --i-table table_se.qza \
  --i-taxonomy taxonomy_se.qza \
  --p-exclude mitochondria,chloroplast \
  --o-filtered-table table_no_mitochondria_chloroplast_se.qza

# Filtering out male samples from host_sex
qiime feature-table filter-samples \
  --i-table table_no_mitochondria_chloroplast_se.qza \
  --m-metadata-file /datasets/project_2/fermentation/fermentation_metadata.tsv \
  --p-where “[host_sex]=’female’” \
  --o-filtered-table female_filtered_table_se.qza

#Filter for female participants
qiime feature-table summarize \
  --i-table female_filtered_table_se.qza \
  --o-visualization female_filtered_table_se.qzv \
  --m-sample-metadata-file /datasets/project_2/fermentation/fermentation_metadata.tsv

# Generate a tree for phylogenetic diversity analyses
qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences rep_seqs_se.qza \
  --o-alignment aligned_rep_seqs_se.qza \
  --o-masked-alignment masked_aligned_rep_seqs_se.qza \
  --o-tree unrooted_tree_se.qza \
  --o-rooted-tree rooted_tree_se.qza 

qiime diversity alpha-rarefaction \
  --i-table table_se.qza \
  --i-phylogeny rooted_tree_se.qza \
  --p-max-depth 62000 \
  --m-metadata-file /datasets/project_2/fermentation/fermentation_metadata.tsv \
  --o-visualization alpha_rarefaction_curve.qzv
